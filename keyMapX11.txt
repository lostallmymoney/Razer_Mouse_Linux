#Minimum loop sleep recommended around 35 ms when interacting with xdotool
#Declare functions, contexts and loops before using them.
#Functions, contexts and loops can use previously declared functions, contexts, and loops.

function=spotifyOpenOnlyOnce
    run=if [ $(pgrep -c spotify) -eq 0 ]; then sudo -Siu $USER bash -c "spotify&"; fi

function=testfunction
    string=Testing, test... Success !!

loop=testloop 
    function=testfunction
    sleep=400

#Negative number forces the loop to run 3 times.
loop=testloop2
    loop=testloop=-3
    sleep=750

context=test
    3 - loop=testloop2

loop=leftClickLoop
    run=xdotool click 1
    sleep=50

loop=rightClickLoop
    run=xdotool click 3
    sleep=50

context=multiupdown
    7 - key=XF86AudioLowerVolume
    9 - key=XF86AudioRaiseVolume

#Nested contexts
context=multimedia
    context=multiupdown
    10 - keyClick=XF86AudioPrev
    12 - keyClick=XF86AudioNext

configWindow=Gnome-terminal
    context=multimedia
    1 - key=XF86AudioMute
    3 - string=gnome@terminal.com
    4 - function=spotifyOpenOnlyOnce
    5 - run=x-terminal-emulator
    6 - specialkey=`
    8 - chmap=Gnome-terminal-Chmap
    11 - key=XF86AudioPlay

config=Gnome-terminal-Chmap
    context=multimedia
    1 - key=XF86AudioMute
    3 - string=gnome@terminal.com Alternative
    4 - function=spotifyOpenOnlyOnce
    5 - userRun=
        x-terminal-emulator -e bash -c '
            echo "=== NAGA SYSTEM INFO ==="
            echo "Host   : $(hostname) | User: $(whoami) | $(uptime -p)"
            echo "CPU    : $(top -bn1 | awk '\''/^%Cpu/{print "User: " $2 "% | Sys: " $4 "% | Idle: " $8 "%"}'\'')"
            echo "Load   : $(cat /proc/loadavg | awk '\''{print "1m: " $1 " | 5m: " $2 " | 15m: " $3}'\'')"
            echo "Memory : $(free -h | awk '\''/^Mem/{print "Used: " $3 " / " $2 " | Available: " $7}'\'')"
            echo "Disk   : $(df -h / | awk '\''NR==2{print "Used: " $3 " / " $2 " | Free: " $4 " (" $5 ")"}'\'')"
            echo "LAN    : $(ip -4 route get 1.1.1.1 2>/dev/null | awk '\''{print $7; exit}'\'')"
            echo "WAN    : $(curl -4 -s https://ifconfig.me || echo "N/A")"
            echo "---"
            echo "Users  : $(w -h 2>/dev/null | awk '\''{print $1 " (" $2 ")"}'\'' | awk '\''BEGIN{ORS=","} {print}'\'' | awk '\''sub(/,$/, "")'\'' || echo "N/A")"
            echo "Logins : $(loginctl list-sessions --no-legend 2>/dev/null | awk '\''{print $1}'\'' | while read sid; do loginctl show-session "$sid" 2>/dev/null | awk -F= '\''/^Name=/{name=$2} /^Timestamp=/{ts=$2} /^Service=/{svc=$2} END{if(name=="'$(whoami)'" && svc~/gdm|login/) print ts " (" name ")"}'\'' ; done | tail -3 | awk '\''BEGIN{ORS=" | "} {print}'\'' | awk '\''sub(/ \| $/, "")'\'')"
            echo "Ports  : $(ss -tuln | tail -n +2 | awk '\''{n=split($5,a,":"); print a[n]}'\'' | grep -E '\''^[0-9]+$'\'' | sort -un | awk '\''BEGIN{ORS=","} {print}'\'' | awk '\''sub(/,$/, "")'\'')"
            read -rsn1 -p "Press any key to close..."
            '
    6 - specialkey=|
    8 - unlockChmap=Gnome-terminal
    11 - key=XF86AudioPlay
