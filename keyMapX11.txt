#Minimum loop sleep recommended around 35 ms when interacting with xdotool
#Declare functions, contexts and loops before using them.
#Functions, contexts and loops can use previously declared functions, contexts, and loops.

function=spotifyOpenOnlyOnce
    run=if [ $(pgrep -c spotify) -eq 0 ]; then sudo -Siu $USER bash -c "spotify&"; fi

function=testfunction
    string=Testing, test... Success !!

loop=testloop 
    function=testfunction
    sleep=400

#Negative number forces the loop to run 3 times.
loop=testloop2
    loop=testloop=-3
    sleep=750

context=test
    3 - loop=testloop2

loop=leftClickLoop
    run=xdotool click 1
    sleep=50

loop=rightClickLoop
    run=xdotool click 3
    sleep=50

context=multiupdown
    7 - key=XF86AudioLowerVolume
    9 - key=XF86AudioRaiseVolume

#Nested contexts
context=multimedia
    context=multiupdown
    10 - keyClick=XF86AudioPrev
    12 - keyClick=XF86AudioNext

# ----> PROFILES HERE <----
